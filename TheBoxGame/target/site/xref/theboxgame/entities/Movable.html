2    * Luokka perii luokan Entity, minkä lisäksi se toteuttaa liikkumiseen tarvittavat metodit.

4   package theboxgame.entities;
5   
6   import java.awt.Graphics;
7   
8   public class Movable extends Entity {
9   
10      public Movable(int x, int y) {
11          super(x, y);
12      }
13  
14      /**
15       * Metodi liikuttaa liikkuvaa oliota yhden Tilen verran vasemmalle.
16       */
17      public void moveLeft() {
18          this.x = this.x - Tile.TILEWIDTH;
19      }
20  
21      /**
22       * Metodi liikuttaa liikkuvaa oliota yhden Tilen verran oikealle.
23       */
24      public void moveRight() {
25          this.x = this.x + Tile.TILEWIDTH;
26      }
27  
28      /**
29       * Metodi liikuttaa liikkuvaa oliota yhden Tilen verran ylöspäin.
30       */
31      public void moveUp() {
32          this.y = this.y - Tile.TILEHEIGHT;
33      }
34  
35      /**
36       * Metodi liikuttaa liikkuvaa oliota yhden Tilen verran alaspäin.
37       */
38      public void moveDown() {
39          this.y = this.y + Tile.TILEHEIGHT;
40      }
41  
42      @Override
43      public void render(Graphics g) {
44  
45      }
46  
47  }

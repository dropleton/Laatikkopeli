6   import java.awt.Color;
7   import java.awt.Graphics;
8   
9   public class Wall extends Entity {
10  
11      private final int id;
12  
13      public Wall(int x, int y) {
14          super(x, y);
15          this.id = 1;
16      }
17  
18      @Override
19      public void render(Graphics g) {
20          g.setColor(Color.black);
21          g.fillRect(x, y, width, height);
22      }
23  
24      @Override
25      public int getId() {
26          return this.id;
27      }
28  
29      @Override
30      public String toString() {
31          return "Wall";
32      }
33  
34  }

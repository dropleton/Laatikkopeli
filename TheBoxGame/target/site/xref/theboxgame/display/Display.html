1   package theboxgame.display;
2   
3   import java.awt.Canvas;
4   import java.awt.Dimension;
5   import javax.swing.JFrame;
6   
7   public class Display {
8   
9       private JFrame frame;
10      private Canvas canvas;
11      private String title;
12      int width, height;
13  
14      public Display(String title, int width, int height) {
15          this.title = title;
16          this.width = width;
17          this.height = height;
18  
19          createDisplay();
20      }
21  
22      private void createDisplay() {
23          frame = new JFrame(title);
24          frame.setSize(this.width, this.height);
25          frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
26          frame.setResizable(false);
27          frame.setLocationRelativeTo(null);
28          frame.setVisible(true);
29          createCanvas();
30          frame.pack();
31      }
32  
33      public void createCanvas() {
34          canvas = new Canvas();
35          canvas.setPreferredSize(new Dimension(width, height));
36          canvas.setMaximumSize(new Dimension(width, height));
37          canvas.setMinimumSize(new Dimension(width, height));
38          canvas.setFocusable(false);
39          frame.add(canvas);
40      }
41  
42      public Canvas getCanvas() {
43          return this.canvas;
44      }
45  
46      public JFrame getJFrame() {
47          return this.frame;
48      }
49  
50  }
